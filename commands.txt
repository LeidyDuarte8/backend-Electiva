# Si ya tienes un repositorio Git inicializado, solo necesitas cambiar el remote
git remote add origin https://github.com/LeidyDuarte8/backend-Electiva.git

# Si tienes cambios para commitear
git add .
git commit -m "Initial commit"

# Subir los cambios a la rama principal
git push -u origin main

# git push -u origin master
